    return distances

–ù—ç–≥ —ç—Ö–ª—ç–ª —Ü—ç–≥—ç—ç—Å –±—É—Å–∞–¥ –±“Ø—Ö —Ü—ç–≥ —Ö“Ø—Ä—Ç—ç–ª —Ö–∞–º–≥–∏–π–Ω –±–æ–≥–∏–Ω–æ –∑–∞–º—ã–≥ –æ–ª.

–ì—Ä–∞—Ñ:

      5
  A ------ B
  | \      |
 2|  9\    |1
  |     \  |
  C ------ D
       6


Nodes: A, B, C, D

Edge weights: A-B=5, A-C=2, A-D=9, B-D=1, C-D=



import heapq  # Priority Queue –∞—à–∏–≥–ª–∞–Ω–∞

# ------------------------------------------------------------
# DIJKSTRA ALGORITHM
# ------------------------------------------------------------
def dijkstra(graph, start):
    # 1Ô∏è‚É£ –ë“Ø—Ö –∑–∞–Ω–≥–∏–ª–∞–∞–Ω—ã –∑–∞–π–≥ ‚àû –≥—ç–∂ –æ–Ω–æ–æ—Ö
    distances = {node: float('inf') for node in graph}
    distances[start] = 0  # –≠—Ö–ª—ç—Ö —Ü—ç–≥–∏–π–Ω –∑–∞–π = 0

    # 2Ô∏è‚É£ Priority queue –±—ç–ª–¥—ç—Ö (—Ö–∞–º–≥–∏–π–Ω –±–∞–≥–∞ –∑–∞–π—Ç–∞–π –∑–∞–Ω–≥–∏–ª–∞–∞–≥ –∞–≤—Ç–æ–º–∞—Ç–∞–∞—Ä –≥–∞—Ä–≥–∞–Ω–∞)
    queue = [(0, start)]  # (–∑–∞–π, –∑–∞–Ω–≥–∏–ª–∞–∞)

    # 3Ô∏è‚É£ –î–∞–≤—Ç–∞–ª—Ç ‚Äì –±“Ø—Ö –∑–∞–Ω–≥–∏–ª–∞–∞–≥–∞–∞—Ä
    while queue:
        current_distance, current_node = heapq.heappop(queue)

        # 4Ô∏è‚É£ –•—ç—Ä–≤—ç—ç –∏–ª“Ø“Ø —É—Ä—Ç –∑–∞–º –±–∞–π–≤–∞–ª –∞–ª–≥–∞—Å–Ω–∞
        if current_distance > distances[current_node]:
            continue

        # 5Ô∏è‚É£ –û–¥–æ–æ–≥–∏–π–Ω –∑–∞–Ω–≥–∏–ª–∞–∞–Ω—ã —Ö”©—Ä—à“Ø“Ø–¥–∏–π–≥ —à–∞–ª–≥–∞—Ö
        for neighbor, weight in graph[current_node].items():
            distance = current_distance + weight

            # 6Ô∏è‚É£ –•—ç—Ä–≤—ç—ç —à–∏–Ω—ç –∑–∞–π –±–æ–≥–∏–Ω–æ –±–∞–π–≤–∞–ª —à–∏–Ω—ç—á–∏–ª–Ω—ç
            if distance < distances[neighbor]:
                distances[neighbor] = distance
                heapq.heappush(queue, (distance, neighbor))

    return distances

# ------------------------------------------------------------
# –ì—Ä–∞—Ñ “Ø“Ø—Å–≥—ç—Ö
# ------------------------------------------------------------
graph = {
    'A': {'B':5, 'C':2, 'D':9},
    'B': {'A':5, 'D':1},
    'C': {'A':2, 'D':6},
    'D': {'A':9, 'B':1, 'C':6}
}

# ------------------------------------------------------------
# –¢—É—Ä—à–∏—Ö —Ö—ç—Å—ç–≥
# ------------------------------------------------------------
start_node = 'A'
shortest_paths = dijkstra(graph, start_node)

print(f"Dijkstra Algorithm “Ø—Ä –¥“Ø–Ω ({start_node}-—Å):")
for node, distance in shortest_paths.items():
    print(f"{start_node} ‚Üí {node} = {distance}")
